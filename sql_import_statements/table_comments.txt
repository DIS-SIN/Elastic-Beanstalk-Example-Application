CREATE TABLE comments(
	course_title_en VARCHAR(200),
	course_code VARCHAR(20),
	survey_id VARCHAR(15),
	short_question VARCHAR(60),
	text_answer TEXT,
	stars TINYINT
);



LOAD DATA INFILE 'D:\\comments.csv'
INTO TABLE comments
CHARACTER SET UTF8
FIELDS OPTIONALLY ENCLOSED BY '"'
TERMINATED BY ','
LINES TERMINATED BY '\n'
IGNORE 1 LINES;



# Index
CREATE INDEX idx_sq ON comments(short_question);
