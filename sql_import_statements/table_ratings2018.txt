CREATE TABLE ratings2018_19(
	course_title VARCHAR(200),
	course_code VARCHAR(20),
	month VARCHAR(10),
	short_question VARCHAR(60),
	numerical_answer FLOAT,
	count SMALLINT
);



LOAD DATA INFILE 'D:\\ratings2018_19.csv'
INTO TABLE ratings2018_19
FIELDS OPTIONALLY ENCLOSED BY '"'
TERMINATED BY ','
LINES TERMINATED BY '\r\n'
IGNORE 1 LINES;
